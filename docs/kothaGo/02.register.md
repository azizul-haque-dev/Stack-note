# Part 2 — Register
## POST /api/v1/auth/register

---

## Flow

```
Client → POST /auth/register { email, password, firstName }
    ↓
Rate limiter (5 req/min per IP)
    ↓
Zod validation
    ↓
Check email not already taken
    ↓
bcrypt hash password (cost 12)
    ↓
Create user in DB
    ↓
Generate access token (JWT, 15min)
    ↓
Generate refresh token (JWT, 7d) → hash → store in DB
    ↓
Web:    Set refresh token as HttpOnly cookie → return { accessToken, user }
Mobile: Return { accessToken, refreshToken, user }
```

## 1. Shared Middleware

### authenticate.ts

```typescript
// backend/src/middleware/authenticate.ts
import { Request, Response, NextFunction } from 'express';
import jwt from 'jsonwebtoken';

// Extend Express Request so req.user is typed everywhere
export interface AuthRequest extends Request {
  user?: {
    sub: string;
    email: string;
    tokenId?: string;
    family?: string;
    rawToken?: string;
  };
}

// ── Access token guard ────────────────────────────────────────────────────────
export function authenticate(req: AuthRequest, res: Response, next: NextFunction) {
  const authHeader = req.headers.authorization;
  if (!authHeader?.startsWith('Bearer ')) {
    return res.status(401).json({ message: 'No access token provided' });
  }

  const token = authHeader.split(' ')[1];

  try {
    const payload = jwt.verify(token, process.env.JWT_ACCESS_SECRET!) as any;
    req.user = payload;
    next();
  } catch (err: any) {
    const message =
      err.name === 'TokenExpiredError' ? 'Access token expired' : 'Invalid access token';
    return res.status(401).json({ message });
  }
}

// ── Refresh token guard ───────────────────────────────────────────────────────
// Web  → HttpOnly cookie
// Mobile → Authorization: Bearer header
export function authenticateRefresh(req: AuthRequest, res: Response, next: NextFunction) {
  const rawToken =
    req.cookies?.refresh_token ?? req.headers.authorization?.split(' ')[1];

  if (!rawToken) {
    return res.status(401).json({ message: 'No refresh token provided' });
  }

  try {
    const payload = jwt.verify(rawToken, process.env.JWT_REFRESH_SECRET!) as any;
    req.user = { ...payload, rawToken }; // rawToken needed later for bcrypt compare
    next();
  } catch (err: any) {
    const message =
      err.name === 'TokenExpiredError' ? 'Refresh token expired' : 'Invalid refresh token';
    return res.status(401).json({ message });
  }
}
```

import { z } from 'zod';
import { Request, Response, NextFunction } from 'express';

export const registerSchema = z.object({
  email: z.string().email('Invalid email format'),
  password: z
    .string()
    .min(8, 'Password must be at least 8 characters')
    .max(72, 'Password too long'), // bcrypt silently truncates beyond 72 bytes — cap it here
  firstName: z.string().min(1).max(50).trim(),
  lastName: z.string().max(50).trim().optional(),
});

export const loginSchema = z.object({
  email: z.string().email(),
  password: z.string().min(1),
});

export const googleAuthSchema = z.object({
  idToken: z.string().min(1),
});

export const forgotPasswordSchema = z.object({
  email: z.string().email(),
});

export const resetPasswordSchema = z.object({
  token: z.string().min(1),
  newPassword: z.string().min(8).max(72),
});

// Middleware factory — validates req.body against a Zod schema
export function validate(schema: z.ZodTypeAny) {
  return (req: Request, res: Response, next: NextFunction) => {
    const result = schema.safeParse(req.body);
    if (!result.success) {
      return res.status(400).json({
        message: 'Validation failed',
        errors: result.error.flatten().fieldErrors,
      });
    }
    req.body = result.data; // Replace body with parsed + stripped data
    next();
  };
}

## 3. Tokens Service

```typescript
// backend/src/tokens/tokens.service.ts
import jwt from 'jsonwebtoken';
import * as bcrypt from 'bcrypt';
import { v4 as uuidv4 } from 'uuid';
import { prisma } from '../lib/prisma';

export interface JwtPayload {
  sub: string;
  email: string;
}

export interface RefreshJwtPayload extends JwtPayload {
  tokenId: string; // Stored as the PK in refresh_tokens collection
  family: string;
}

export const tokensService = {
  // ─── Access Token ───────────────────────────────────────────────────────────
  generateAccessToken(payload: JwtPayload): string {
    return jwt.sign(payload, process.env.JWT_ACCESS_SECRET!, {
      expiresIn: (process.env.JWT_ACCESS_EXPIRY ?? '15m') as any,
    });
  },

  // ─── Refresh Token ──────────────────────────────────────────────────────────
  async generateRefreshToken(
    userId: string,
    email: string,
    family: string,
    meta: { userAgent?: string; ipAddress?: string } = {},
  ): Promise<string> {
    const tokenId = uuidv4();
    const expiryDays = parseInt(process.env.JWT_REFRESH_EXPIRY_DAYS ?? '7');

    const payload: RefreshJwtPayload = { sub: userId, email, tokenId, family };

    const rawToken = jwt.sign(payload, process.env.JWT_REFRESH_SECRET!, {
      expiresIn: `${expiryDays}d` as any,
    });

    // Hash before storing. Cost 10 is fine here — tokens are random, no dict attack risk.
    const tokenHash = await bcrypt.hash(rawToken, 10);

    const expiresAt = new Date();
    expiresAt.setDate(expiresAt.getDate() + expiryDays);

    await prisma.refreshToken.create({
      data: { id: tokenId, tokenHash, family, userId, expiresAt, ...meta },
    });

    return rawToken;
  },

  // ─── Rotate (see Part 4) ────────────────────────────────────────────────────
  async rotateRefreshToken(
    payload: RefreshJwtPayload & { rawToken: string },
    meta: { userAgent?: string; ipAddress?: string } = {},
  ): Promise<{ accessToken: string; refreshToken: string }> {
    const storedToken = await prisma.refreshToken.findUnique({
      where: { id: payload.tokenId },
    });

    if (!storedToken) {
      await tokensService.revokeTokenFamily(payload.family);
      const err: any = new Error('Refresh token reuse detected. All sessions revoked.');
      err.status = 403;
      throw err;
    }

    if (storedToken.isRevoked) {
      await tokensService.revokeTokenFamily(payload.family);
      const err: any = new Error('Revoked token used. All sessions revoked.');
      err.status = 403;
      throw err;
    }

    if (new Date() > storedToken.expiresAt) {
      const err: any = new Error('Refresh token expired');
      err.status = 401;
      throw err;
    }

    const isValid = await bcrypt.compare(payload.rawToken, storedToken.tokenHash);
    if (!isValid) {
      await tokensService.revokeTokenFamily(payload.family);
      const err: any = new Error('Invalid refresh token');
      err.status = 403;
      throw err;
    }

    await prisma.refreshToken.delete({ where: { id: payload.tokenId } });

    const newAccessToken = tokensService.generateAccessToken({
      sub: payload.sub,
      email: payload.email,
    });

    const newRefreshToken = await tokensService.generateRefreshToken(
      payload.sub,
      payload.email,
      payload.family,
      meta,
    );

    return { accessToken: newAccessToken, refreshToken: newRefreshToken };
  },

  async revokeTokenFamily(family: string): Promise<void> {
    await prisma.refreshToken.updateMany({
      where: { family },
      data: { isRevoked: true },
    });
  },

  async revokeUserTokens(userId: string, tokenId?: string): Promise<void> {
    if (tokenId) {
      await prisma.refreshToken.deleteMany({ where: { id: tokenId, userId } });
    } else {
      await prisma.refreshToken.updateMany({
        where: { userId },
        data: { isRevoked: true },
      });
    }
  },

  async cleanExpiredTokens(): Promise<void> {
    await prisma.refreshToken.deleteMany({
      where: { expiresAt: { lt: new Date() } },
    });
  },
};
```

## 4. Auth Service — register()

```typescript
// backend/src/auth/auth.service.ts  (register portion)
import * as bcrypt from 'bcrypt';
import { AuthProvider } from '@prisma/client';
import { v4 as uuidv4 } from 'uuid';
import { usersService } from '../users/users.service';
import { tokensService } from '../tokens/tokens.service';

type Meta = { userAgent?: string; ipAddress?: string };

export function sanitizeUser(user: any) {
  const { passwordHash, ...safe } = user;
  return safe;
}

export const authService = {
  async register(
    data: { email: string; password: string; firstName: string; lastName?: string },
    meta: Meta = {},
  ) {
    // 1. Check for duplicate email
    const exists = await usersService.findByEmail(data.email);
    if (exists) {
      const err: any = new Error('Email already registered');
      err.status = 409;
      throw err;
    }

    // 2. Hash password
    // Cost 12 = ~300ms on modern hardware. Balances security vs UX.
    const saltRounds = parseInt(process.env.BCRYPT_SALT_ROUNDS ?? '12');
    const passwordHash = await bcrypt.hash(data.password, saltRounds);

    // 3. Create user
    const user = await usersService.create({
      email: data.email.toLowerCase().trim(),
      passwordHash,
      firstName: data.firstName,
      lastName: data.lastName,
      provider: AuthProvider.local,
    });

    // 4. Issue tokens
    const family = uuidv4(); // Brand new family for this session
    const accessToken = tokensService.generateAccessToken({ sub: user.id, email: user.email });
    const refreshToken = await tokensService.generateRefreshToken(
      user.id, user.email, family, meta,
    );

    return { accessToken, refreshToken, user: sanitizeUser(user) };
  },

  // login, googleAuth, etc. — see Part 3
};
```

## 5. Auth Router — /register route

```typescript
// backend/src/auth/auth.router.ts  (register portion)
import { Router, Request, Response, NextFunction } from 'express';
import { authService } from './auth.service';
import { validate, registerSchema } from './auth.validators';
import { registerLimiter } from '../middleware/rate-limit';

const router = Router();

const IS_PRODUCTION = process.env.NODE_ENV === 'production';

export const REFRESH_COOKIE_OPTIONS = {
  httpOnly: true,                  // JS cannot read — blocks XSS token theft
  secure: IS_PRODUCTION,           // HTTPS only in prod
  sameSite: 'strict' as const,     // Blocks CSRF from other origins
  maxAge: 7 * 24 * 60 * 60 * 1000,
  path: '/api/v1/auth',            // Cookie only sent on auth routes, not your whole API
};

export function isMobile(req: Request): boolean {
  const ua = req.headers['user-agent'] ?? '';
  return /ReactNative|Expo|okhttp/.test(ua);
}

export function getMeta(req: Request) {
  return {
    userAgent: req.headers['user-agent'],
    ipAddress: (req.headers['x-forwarded-for'] as string) ?? req.ip,
  };
}

// POST /api/v1/auth/register
router.post(
  '/register',
  registerLimiter,
  validate(registerSchema),
  async (req: Request, res: Response, next: NextFunction) => {
    try {
      const { accessToken, refreshToken, user } = await authService.register(
        req.body,
        getMeta(req),
      );

      // Web: refresh token goes in HttpOnly cookie, never in response body
      if (!isMobile(req)) {
        res.cookie('refresh_token', refreshToken, REFRESH_COOKIE_OPTIONS);
        return res.status(201).json({ accessToken, user });
      }

      // Mobile: refresh token in response body — stored in Keychain/EncryptedStorage
      return res.status(201).json({ accessToken, refreshToken, user });
    } catch (err) {
      next(err);
    }
  },
);

export { router as authRouter };
```
