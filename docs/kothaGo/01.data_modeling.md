## 2. Prisma Schema

```prisma
// backend/prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "mongodb"
  url      = env("DATABASE_URL")
}

// ─── Enums ────────────────────────────────────────────────────────────────────

enum AuthProvider {
  local
  google
}

// ─── User ─────────────────────────────────────────────────────────────────────

model User {
  id            String               @id @default(auto()) @map("_id") @db.ObjectId
  email         String               @unique
  passwordHash  String?              // null for Google-only users
  firstName     String?
  lastName      String?
  avatarUrl     String?
  provider      AuthProvider         @default(local)
  googleId      String?              @unique
  emailVerified Boolean              @default(false)
  isActive      Boolean              @default(true)

  refreshTokens    RefreshToken[]
  passwordResets   PasswordResetToken[]

  createdAt     DateTime             @default(now())
  updatedAt     DateTime             @updatedAt

  @@map("users")
}

// ─── Refresh Token ────────────────────────────────────────────────────────────

model RefreshToken {
  id        String   @id @map("_id") @db.ObjectId
  // ↑ We set this manually to the JWT tokenId (uuidv4) — NOT auto-generated.
  // This is intentional: we use it to look up the exact token record for rotation.

  // Stored HASHED — bcrypt cost 10.
  // Never store the raw token. If DB is breached, hashed tokens are useless.
  tokenHash String

  // Token family: all rotations of the same login session share one family UUID.
  // Reuse of a rotated token → revoke entire family (theft detection).
  family    String

  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  userId    String   @db.ObjectId

  expiresAt DateTime
  isRevoked Boolean  @default(false)

  // Device metadata for session management dashboard
  userAgent String?
  ipAddress String?

  createdAt DateTime @default(now())

  @@index([family])   // Fast lookup for family-based revocation
  @@index([userId])   // Fast lookup for user session listing
  @@map("refresh_tokens")
}

// ─── Password Reset Token ─────────────────────────────────────────────────────

model PasswordResetToken {
  id        String   @id @default(auto()) @map("_id") @db.ObjectId

  // Stored HASHED — same reason as refresh tokens.
  // Raw token is only ever in the email link, never in the DB.
  tokenHash String

  user      User     @relation(fields: [userId], references: [id], onDelete: Cascade)
  userId    String   @db.ObjectId

  expiresAt DateTime // Short TTL — 15 minutes
  isUsed    Boolean  @default(false)

  createdAt DateTime @default(now())

  @@index([userId])
  @@map("password_reset_tokens")
}
```

> **Commands:**

## 3. Prisma Singleton

```typescript
// backend/src/lib/prisma.ts
import { PrismaClient } from '@prisma/client';

// Without this guard, ts-node-dev's hot reload would create a new PrismaClient
// on every file change, exhausting your MongoDB connection pool fast.
const globalForPrisma = globalThis as unknown as { prisma: PrismaClient };

export const prisma =
  globalForPrisma.prisma ??
  new PrismaClient({
    log: process.env.NODE_ENV === 'development' ? ['query', 'error'] : ['error'],
  });

if (process.env.NODE_ENV !== 'production') {
  globalForPrisma.prisma = prisma;
}
```

## 4. Users Service

```typescript
// backend/src/users/users.service.ts
import { prisma } from '../lib/prisma';
import { Prisma } from '@prisma/client';

export const usersService = {
  async findById(id: string) {
    return prisma.user.findUnique({ where: { id } });
  },

  async findByEmail(email: string) {
    return prisma.user.findUnique({
      where: { email: email.toLowerCase().trim() },
    });
  },

  // passwordHash has no @omit in Prisma — we manually select it only when needed.
  // Never return it from a route. Keeping it explicit here makes it obvious at call sites.
  async findByEmailWithPassword(email: string) {
    return prisma.user.findUnique({
      where: { email: email.toLowerCase().trim() },
      select: {
        id: true,
        email: true,
        passwordHash: true,
        provider: true,
        isActive: true,
        firstName: true,
        lastName: true,
        avatarUrl: true,
        emailVerified: true,
        googleId: true,
        createdAt: true,
        updatedAt: true,
      },
    });
  },

  async findByGoogleId(googleId: string) {
    return prisma.user.findUnique({ where: { googleId } });
  },

  async create(data: Prisma.UserCreateInput) {
    return prisma.user.create({ data });
  },

  async updatePassword(userId: string, passwordHash: string) {
    return prisma.user.update({
      where: { id: userId },
      data: { passwordHash },
    });
  },
};
```

## 5. server.ts & app.ts

```typescript
// backend/src/server.ts
import 'dotenv/config';
import { app } from './app';
import { prisma } from './lib/prisma';

const PORT = process.env.PORT ?? 3001;

async function main() {
  await prisma.$connect();
  console.log('✓ Connected to MongoDB');

  app.listen(PORT, () => {
    console.log(`✓ Auth server running on port ${PORT}`);
  });
}

main().catch(async (err) => {
  console.error(err);
  await prisma.$disconnect();
  process.exit(1);
});
```

```typescript
// backend/src/app.ts
import express, { Request, Response, NextFunction } from 'express';
import helmet from 'helmet';
import cors from 'cors';
import cookieParser from 'cookie-parser';
import { authRouter } from './auth/auth.router';

const app = express();

app.use(helmet());
app.use(cors({
  origin: process.env.WEB_CLIENT_URL,
  credentials: true,
  methods: ['GET', 'POST', 'PUT', 'DELETE', 'OPTIONS'],
  allowedHeaders: ['Content-Type', 'Authorization', 'X-CSRF-Token'],
}));
app.use(express.json({ limit: '10kb' }));
app.use(cookieParser());

app.use('/api/v1/auth', authRouter);

// 404
app.use((_req: Request, res: Response) => {
  res.status(404).json({ message: 'Route not found' });
});

// Global error handler — Express identifies this by the 4-arg signature
app.use((err: any, _req: Request, res: Response, _next: NextFunction) => {
  const status = err.status ?? err.statusCode ?? 500;
  const message = err.message ?? 'Internal server error';
  if (process.env.NODE_ENV !== 'production') console.error(err);
  res.status(status).json({ message });
});

export { app };
```

## 7. Schema Design Decisions

| Decision | Why |
|---|---|
| `RefreshToken.id` set manually (not auto) | We use the JWT `tokenId` claim (uuidv4) as the PK so we can do a single `findUnique({ where: { id: tokenId } })` during rotation — no extra lookup needed |
| `passwordHash` nullable | Google users have no password. Nullable makes the distinction clear in the type system |
| `PasswordResetToken` separate model | Reset tokens have different TTL (15 min), usage semantics (one-time), and lifecycle from refresh tokens — mixing them would be wrong |
| `family` on RefreshToken | Enables wiping an entire session lineage on theft detection without knowing all individual token IDs |
| `@@index([family])` | Without this, revoking a family does a full collection scan — unacceptable at scale |
